Fluence DB Example


  • CREATE TABLE users(id int, name varchar(128), age int)
  • INSERT INTO users VALUES(1, 'Sara', 23)
  • INSERT INTO users VALUES(2, 'Bob', 19), (3, 'Caroline', 31), (4, 'Max', 25)
  • SELECT MAX(age) FROM users
  • SELECT AVG(age) FROM users
  • SELECT COUNT(*) FROM users
  • UPDATE users SET age = 20 WHERE id = 2
  • SELECT * FROM users WHERE name = 'unknown'
  • SELECT id, name FROM users
  • EXPLAIN SELECT id, name FROM users
  • CREATE TABLE roles(user_id int, role varchar(128))
  • INSERT INTO roles VALUES(1, 'Teacher'), (2, 'Student'), (3, 'Scientist'), (4, 'Writer')
  • SELECT u.name AS Name, r.role AS Role FROM users u JOIN roles r ON u.id = r.user_id WHERE r.role = 'Writer'
  • SELECT salary FROM USERS
